<Heading level="2">
    Readable Version (stepB_node.js)
</Heading>

<Paragraph blockColor="light_grey">
<Paragraph>
    <Mark>function EVAL(env, ast, f, el) {</Mark>
</Paragraph>
<Paragraph>
    <Mark>    while (true) {</Mark>
</Paragraph>
<Paragraph>
    <Mark>        if (!Array.isArray(ast)) {</Mark>
</Paragraph>
<Paragraph>
    <Mark>            if (typeof ast == "string") {</Mark>
</Paragraph>
<Paragraph>
    <Mark>                return ast in env ? env[ast] : E.throw(ast + " not found");</Mark>
</Paragraph>
<Paragraph>
    <Mark>            } else {</Mark>
</Paragraph>
<Paragraph>
    <Mark>                return ast ?? null;</Mark>
</Paragraph>
<Paragraph>
    <Mark>            }</Mark>
</Paragraph>
<Paragraph>
    <Mark>        } else {</Mark>
</Paragraph>
<Paragraph>
    <Mark>            // Apply forms...</Mark>
</Paragraph>
<Paragraph>
    <Mark>        }</Mark>
</Paragraph>
<Paragraph>
    <Mark>    }</Mark>
</Paragraph>
<Paragraph>
    <Mark>}</Mark>
</Paragraph>
</Paragraph>

<Heading level="3">
    new_env - Environment Creation
</Heading>

<Paragraph blockColor="light_grey">
<Paragraph>
    <Mark>function new_env(env, binds, exprs) {</Mark>
</Paragraph>
<Paragraph>
    <Mark>    env = Object.create(env);</Mark>
</Paragraph>
<Paragraph>
    <Mark>    binds.some((b,i) => b == "&" ? </Mark>
</Paragraph>
<Paragraph>
    <Mark>        env[binds[i+1]] = exprs.slice(i) :</Mark>
</Paragraph>
<Paragraph>
    <Mark>        (env[b] = exprs[i], 0));</Mark>
</Paragraph>
<Paragraph>
    <Mark>    return env;</Mark>
</Paragraph>
<Paragraph>
    <Mark>}</Mark>
</Paragraph>
</Paragraph>

<Paragraph>
    Creates a new environment with prototype chain for scope lookup.
</Paragraph>
